Agregar password del MySQL y guardar como .env

PUERTO = 3000

HOST = 'localhost'

MYSQLPORT = 3306

MYSQLUSER = 'root'

MYSQLPASS = ''

DATABASE = 'reclamos'

Ejecutar con:
    NPM RUN start : "node ./src/app.js",
    NPM RUN dev   : "nodemon ./src/app.js",
    NPM RUN watch : "node --watch ./src/app.js",
